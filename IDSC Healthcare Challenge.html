# =========================
# BUILD DATASET FROM ECG
# =========================
import numpy as np

X = []
y = []

for _, row in metadata.iterrows():
    pid = str(row["patient_id"])

    record = wfdb.rdrecord(f"files1/{pid}/{pid}")
    signals = record.p_signal
    lead_names = record.sig_name

    v1 = lead_names.index("V1")
    v2 = lead_names.index("V2")
    v3 = lead_names.index("V3")

    selected = signals[:, [v1, v2, v3]]   # shape (1200, 3)
    X.append(selected.flatten())           # shape (3600,)
    y.append(row["brugada"])

X = np.array(X)
y = np.array(y)

print("X shape:", X.shape)
print("y shape:", y.shape)

y shape: (363,)
